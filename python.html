អំពី Python

Python គឺជាភាសាកម្មវិធីមួយដែលមានប្រជាប្រិយភាពសម្រាប់ការអភិវឌ្ឍន៍កម្មវិធី និងការវិភាគទិន្នន័យ។ វាជាសិល្បៈនៃការកំណត់និងដំណើរការប្រព័ន្ធមួយដោយប្រើលទ្ធផល និង OOP។

មូលដ្ឋាន Python

កូដដំបូង

យើងចាប់ផ្ដើម Python ដោយប្រើ print()។

print("សួស្តី ពិភពលោក!")

លទ្ធផល:

សួស្តី ពិភពលោក!

ប្រភេទទិន្នន័យ និងអថេរ

ប្រភេទទិន្នន័យមានដូចជា int, float, str ។ ឧទាហរណ៍៖

age = 25
height = 1.75
name = "Chantha"
print(f"ឈ្មោះ: {name}, អាយុ: {age}, កម្ពស់: {height} ម៉ែត្រ")

លទ្ធផល:

ឈ្មោះ: Chantha, អាយុ: 25, កម្ពស់: 1.75 ម៉ែត្រ

ការបញ្ចូល និងបង្ហាញទិន្នន័យ

ប្រើ input() ដើម្បីបញ្ចូលទិន្នន័យ និង print() ដើម្បីបង្ហាញទិន្នន័យ។

age = int(input("បញ្ចូលអាយុរបស់អ្នក: "))
print(f"អាយុរបស់អ្នកគឺ {age}")

លទ្ធផល:

បញ្ចូលអាយុរបស់អ្នក: 30
អាយុរបស់អ្នកគឺ 30

ប្រមាណវិធី និង ប្រមាណវិធីផ្សេងៗ

ប្រមាណវិធីគណនា (Arithmetic Operators)

ប្រមាណវិធីសម្រាប់គណនា: +, -, *, /, %, **, //.

a = 10
            b = 3
            print("ផលបូក:", a + b)         # Addition
            print("ផលគុណ:", a * b)          # Multiplication
            print("ផ្លូវចែក:", a / b)        # Division
            print("ចែកសម្រាប់បច្ចុប្បន្ន:", a // b)  # Floor Division
            print("សំណល់:", a % b)          # Modulus
            print("ថាមពល:", a ** b)         # Exponentiation

លទ្ធផល:

ផលបូក: 13
            ផលគុណ: 30
            ផ្លូវចែក: 3.3333333333333335
            ចែកសម្រាប់បច្ចុប្បន្ន: 3
            សំណល់: 1
            ថាមពល: 1000

ប្រមាណវិធីប្រៀបធៀប (Comparison Operators)

ប្រមាណវិធីសម្រាប់ប្រៀបធៀប: ==, !=, >, <, >=, <=.

a = 10
            b = 5
            print("អតិថិជនដូចគ្នា:", a == b)
            print("អតិថិជនមិនដូចគ្នា:", a != b)
            print("អតិថិជនធំជាង:", a > b)
            print("អតិថិជនតិចជាង:", a < b)
            print("អតិថិជនធំឬស្មើ:", a >= b)
            print("អតិថិជនតិចឬស្មើ:", a <= b)

លទ្ធផល:

អតិថិជនដូចគ្នា: False
            អតិថិជនមិនដូចគ្នា: True
            អតិថិជនធំជាង: True
            អតិថិជនតិចជាង: False
            អតិថិជនធំឬស្មើ: True
            អតិថិជនតិចឬស្មើ: False

ប្រមាណវិធីទំនាក់ទំនង (Logical Operators)

ប្រមាណវិធីសម្រាប់ទំនាក់ទំនង: and, or, not.

a = True
            b = False
            print("a និង b:", a and b)
            print("a ឬ b:", a or b)
            print("មិន a:", not a)

លទ្ធផល:

a និង b: False
            a ឬ b: True
            មិន a: False

ប្រមាណវិធីបញ្ជា (Assignment Operators)

ប្រមាណវិធីសម្រាប់បញ្ជា: =, +=, -=, *=, /=, %=.

a = 5
            a += 3   # a = a + 3
            print("a បន្ទាប់ពីបន្ថែម 3:", a)
            a *= 2   # a = a * 2
            print("a បន្ទាប់ពីគុណ 2:", a)

លទ្ធផល:

a បន្ទាប់ពីបន្ថែម 3: 8
            a បន្ទាប់ពីគុណ 2: 16

ប្រមាណវិធីប៊ីត (Bitwise Operators)

ប្រមាណវិធីសម្រាប់ប្រើប្រាស់ប្រមាណវិធីប៊ីត: &, |, ^, ~, <<, >>.

a = 10  # 1010 in binary
            b = 4   # 0100 in binary
            print("AND:", a & b)     # 0
            print("OR:", a | b)      # 14
            print("XOR:", a ^ b)     # 14
            print("NOT:", ~a)        # -11
            print("Left Shift:", a << 1)  # 20
            print("Right Shift:", a >> 1) # 5

លទ្ធផល:

AND: 0
            OR: 14
            XOR: 14
            NOT: -11
            Left Shift: 20
            Right Shift: 5

ប្រមាណវិធីសម្គាល់ (Identity Operators)

ប្រមាណវិធីសម្រាប់សម្គាល់: is, is not.

a = [1, 2, 3]
            b = a
            c = a[:]
            print("a ជាមួយ b:", a is b)      # True
            print("a ជាមួយ c:", a is c)      # False
            print("a មិនមែនជាមួយ c:", a is not c)  # True

លទ្ធផល:

a ជាមួយ b: True
            a ជាមួយ c: False
            a មិនមែនជាមួយ c: True

ប្រមាណវិធីសមាជិក (Membership Operators)

ប្រមាណវិធីសម្រាប់សមាជិក: in, not in.

fruits = ["ប៉ោម", "បង្គា", "អំពិល"]
            print("ប៉ោមនៅក្នុងផ្ទាំងទេ:", "ប៉ោម" in fruits)   # True
            print("ស្រទូសមិននៅក្នុងផ្ទាំងទេ:", "ស្រទូស" not in fruits)  # True

លទ្ធផល:

ប៉ោមនៅក្នុងផ្ទាំងទេ: True
            ស្រទូសមិននៅក្នុងផ្ទាំងទេ: True

គ្រប់គ្រងលូប និងលក្ខខណ្ឌ

If Statement

ប្រើ if ដើម្បីត្រួតពិនិត្យលទ្ធផល

score = 90
if score >= 80:
    print("ជាប់!")
else:
    print("សូមព្យាយាមម្តងទៀត!")

លទ្ធផល:

ជាប់!

Switch

Python គ្មានការប្រើប្រាស់ switch តែអាចប្រើ if-elif-else ផ្ទុយទៅវិញ។

day = 3

if day == 1:
    print("ថ្ងៃច័ន្ទ")
elif day == 2:
    print("ថ្ងៃអង្គារ")
elif day == 3:
    print("ថ្ងៃពុធ")
elif day == 4:
    print("ថ្ងៃព្រហស្បតិ៍")
elif day == 5:
    print("ថ្ងៃសុក្រ")
elif day == 6:
    print("ថ្ងៃសៅរ៍")
elif day == 7:
    print("ថ្ងៃអាទិត្យ")
else:
    print("មិនមែនជាថ្ងៃដែលត្រឹមត្រូវទេ")

For Loop

បញ្ជី for ត្រូវបានប្រើសម្រាប់កំណត់ចំនួនតំណរ ដើម្បីអនុវត្តនូវកូដលើធាតុគ្រប់សំរាប់ក្នុងកំណត់។ នៅក្នុងករណីនេះ range(5) មានន័យថានឹងចាប់ផ្តើមពី ០ ដល់ ៤ (មិនរាប់ ៥)។

for i in range(5):
    print(i)

1
2

While Loop

បញ្ជី while អនុញ្ញាតឱ្យអ្នកអនុវត្តនូវកូដរហូតដល់លក្ខខណ្ឌមួយត្រូវបានបំពេញ។ នៅក្នុងករណីនេះ count < 5 មែនទែននឹងធ្វើឱ្យប្រើប្រាស់បន្ទាត់ print(count) រាល់ពេលដែល count តិចជាង ៥។

count = 0
while count < 5:
    print(count)
    count += 1

Do While Loop

Python មិនមានបញ្ជី do while ដែលបានកំណត់ទេ ប៉ុន្តែអ្នកអាចសមីការពីវាប្រើប្រាស់ while បញ្ជី។ ក្នុង do while បញ្ជី ការចាប់ផ្ដើមនៃកូដត្រូវបានអនុវត្តយ៉ាងតិចមួយលើសពីមុន មុនពេលដែលលក្ខណ្ឌត្រូវបានពិនិត្យ។

count = 0
while True:
    print(count)
    count += 1
    if count >= 5:
        break

មុខងារ (Function)

មុខងារដែលត្រឡប់ (Returning Functions)

មុខងារ​ដែល​ត្រឡប់​តម្លៃ:

def add(a, b):
            return a + b
        
        result = add(5, 3)
        print(result)  # លទ្ធផល: 8

មុខងារដែលមិនត្រឡប់ (Non-Returning Functions)

មុខងារ​ដែល​មិន​ត្រឡប់​តម្លៃ:

def greet(name):
            print(f"សួស្តី {name}!")
        
        greet("Chantha")  # លទ្ធផល: សួស្តី Chantha!

មុខងារដែលមានអារម្មណ៍ (Functions with Parameters)

មុខងារដែលមានអារម្មណ៍:

def multiply(a, b):
            return a * b
        
        result = multiply(4, 5)
        print(result)  # លទ្ធផល: 20

មុខងារដែលគ្មានអារម្មណ៍ (Functions without Parameters)

មុខងារដែលគ្មានអារម្មណ៍:

def display_message():
            print("Hello, World!")
        
        display_message()  # លទ្ធផល: Hello, World!

អារេ (Array)

អារេនៅក្នុង Python ប្រើប្រាស់សម្រាប់រក្សាទុកធាតុជាច្រើន។

អារេតែមួយ (Single-Dimensional Array)

អារេតែមួយគឺជាការប្រមូលផ្តុំរបស់ធាតុទាំងអស់ដែលមានប្រភេទដូចគ្នា។

fruits = ["ប៉ោម", "ខេមរ", "អណ្ដូង"]
print(fruits[0])  # បង្ហាញធាតុដំបូង

លទ្ធផល:

ប៉ោម

អារេច្រើន (Multi-Dimensional Array)

អារេច្រើនគឺជាអារេដែលមានអារេផ្សេងទៀតនៅក្នុងវា។

matrix = [[1, 2, 3], [4, 5, 6], [7, 8, 9]]
print(matrix[1][2])  # បង្ហាញធាតុក្នុងជួរ 1 និងជួរ 2

លទ្ធផល:

6

Object-Oriented Programming (OOP)

កម្មវិធីមើលមុខវត្ថុ (OOP) គឺជាវិធីសាស្ត្រដើម្បីអភិវឌ្ឍកម្មវិធីដោយប្រើ class និង object។ វាមានលក្ខណៈសំខាន់ៗដែលរួមមាន:

Class និង Object

Class គឺជាប្រភេទទិន្នន័យដែលអ្នកបង្កើតឡើង ហើយ Object គឺជាករណីនៃ Class។

class Dog:
    def __init__(self, name):
        self.name = name

    def bark(self):
        return "Woof!"

my_dog = Dog("Buddy")
print(my_dog.name)
print(my_dog.bark())

លទ្ធផល:

Buddy
Woof!

(Inheritance)

Inheritance អនុញ្ញាតឱ្យ Class ថ្មីទទួលសេរីភាពពី Class ដែលមានស្រាប់។

class Animal:
    def speak(self):
        return "Some sound"

class Cat(Animal):
    def speak(self):
        return "Meow!"

my_cat = Cat()
print(my_cat.speak())

លទ្ធផល:

Meow!

(Polymorphism)

Polymorphism អនុញ្ញាតឱ្យប្រើប្រាស់មុខងារដូចគ្នានៅក្នុង Class ច្រើន។

def animal_sound(animal):
    print(animal.speak())

animal_sound(my_dog)  # Dog
animal_sound(my_cat)  # Cat

លទ្ធផល:

Woof!
Meow!

(Encapsulation)

Encapsulation គឺជាការបង្រួមទិន្នន័យ និងមុខងារ។ វាជួយក្នុងការពារ និងគ្រប់គ្រងអត្ថបទ។

class BankAccount:
    def __init__(self, balance=0):
        self.__balance = balance  # Private attribute

    def deposit(self, amount):
        self.__balance += amount

    def get_balance(self):
        return self.__balance

account = BankAccount()
account.deposit(100)
print(account.get_balance())

លទ្ធផល:

100

(Abstraction)

Abstraction គឺជាការបង្ហាញផែនការអោយអ្នកប្រើដោយមិនបង្ហាញអ្វីដែលកើតឡើងនៅក្នុងដំណើរការនោះ។

from abc import ABC, abstractmethod

class Shape(ABC):
    @abstractmethod
    def area(self):
        pass

class Rectangle(Shape):
    def __init__(self, width, height):
        self.width = width
        self.height = height

    def area(self):
        return self.width * self.height

rectangle = Rectangle(5, 10)
print(rectangle.area())

លទ្ធផល:

50

# សរសេរ​ទិន្នន័យទៅឯកសារ
    with open("output.txt", "w") as file:

អាដេកទ័រ (Decorators)

Decorators គឺជាមុខងារដែលអាចបន្ថែមអ្វីមួយទៅមុខងារផ្សេង។

def decorator_function(original_function):
        def wrapper_function():
            print("Decorator executed this before {}".format(original_function.__name__))
            return original_function()
        return wrapper_function
    
    @decorator_function
    def display():
        return "Display function executed!"
    
    print(display())

លទ្ធផល:

Decorator executed this before display
    Display function executed!

សំណួរ និង ចម្លើយ Python

1. Python គឺអ្វី?
Python គឺជាសាស្ត្រាសំរាប់កម្មវិធី និងអភិវឌ្ឍន៍ដែលមានសមត្ថភាពខ្ពស់។ វាត្រូវបានប្រើសម្រាប់ការអភិវឌ្ឍន៍វេបសាយ, អង់ត្រូវនិងកម្មវិធី, វិទ្យាសាស្ត្រ និងការវិភាគទិន្នន័យ។
2. តើ Python គាំទ្រដល់ OOP ឬអត់?
បាទ, Python គាំទ្រទៅ OOP (កម្មវិធីមួយគ្នា) ដូចជាការបង្កើត class និង object។
3. តើតម្រុយខ្លះដែល Python មាន?
Python មានតម្រុយគ្នាច្រើនដូចជា: + (បូក), - (កាត់), * (គុណ), / (ចែក) និង % (សំណល់)។
4. តើ Python អាចប្រើប្រាស់សម្រាប់អ្វីខ្លះ?
Python អាចប្រើប្រាស់សម្រាប់ការអភិវឌ្ឍន៍វេបសាយ, វិស្វកម្ម, សិក្សាគណិតវិទ្យា, ស្រាវជ្រាវវិទ្យាសាស្ត្រ, និងការវិភាគទិន្នន័យ។
5. តើ Python មានមុខងារនៅក្នុងការចងក្រងទិន្នន័យយ៉ាងដូចម្តេច?
Python មានមុខងារច្រើនដូចជា list, tuple, dictionary និង set ដែលអាចប្រើប្រាស់ក្នុងការចងក្រងទិន្នន័យ។
6. តើ Python មានគណនីផងដែរឬទេ?
បាទ, Python អាចធ្វើការគណនាទាំងស្តង់ដារនិងទាំងខ្ពស់ដោយប្រើប្រាស់ប្រមាណវិធីផ្សេងៗ។
7. តើ Python អាចធ្វើការអភិវឌ្ឍន៍វេបសាយយ៉ាងដូចម្តេច?
Python អាចប្រើប្រាស់សម្រាប់ការអភិវឌ្ឍន៍វេបសាយដោយប្រើប្រាស់ម៉ូឌុលដូចជា Flask និង Django ដែលអនុញ្ញាតឱ្យបង្កើតកម្មវិធីវេបសាយបានយ៉ាងងាយស្រួល។
8. តើ Python មានការគាំទ្រដល់បណ្ណាល័យផ្សេងៗទៀតឬទេ?
បាទ, Python មានការគាំទ្របណ្ណាល័យជាច្រើនដូចជា NumPy សម្រាប់ការគណនា, Pandas សម្រាប់ការវិភាគទិន្នន័យ, និង Matplotlib សម្រាប់ការធ្វើក្រាហ្វិក។
9. តើ Python មានប្រភេទទិន្នន័យណាខ្លះ?
Python មានប្រភេទទិន្នន័យដូចជា int (ចំនួនគត់), float (ចំនួនទឹក), str (អក្សរ), និង bool (តម្លៃពិត/ក្លែងក្លាយ)។
10. តើខ្ញុំអាចស្វែងយល់ពី Python នៅឯណា?
អ្នកអាចស្វែងយល់ពី Python តាមរយៈវេបសាយសិក្សាផ្សេងៗ, ការបណ្ដុះបណ្ដាល, សៀវភៅ និងវីដេអូសិក្សា។
